class Solution {
public:
    bool validPalindrome(string s) {
        
        for (int i = 0, j = s.size() - 1; i < j; ++i, --j) {
            if (s[i] != s[j]) {
                int low1 = i + 1, high1 = j;
                int low2 = i, high2 = j - 1;
                while (low1 <= high1 && s[low1] == s[high1]) {
                    low1++;
                    high1--;
                }
                while (low2 <= high2 && s[low2] == s[high2]) {
                    low2++;
                    high2--;
                }
                return low1 > high1 || low2 > high2;
            }
        }
        return true;
    }
};
