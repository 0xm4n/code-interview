class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> res;
        postorder(root, res);
        return res;
    }
    
    void postorder(TreeNode* root, vector<int>& arr) {
        if (!root) return;
        if (root->left) postorder(root->left, arr);
        if (root->right) postorder(root->right, arr);
        arr.push_back(root->val);
    }
};
