class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        inorder(root, res);
        return res;
        
    }
    
    void inorder(TreeNode* root, vector<int>& arr) {
        if (!root) return;
        if (root->left) inorder(root->left, arr);
        arr.push_back(root->val);
        if (root->right) inorder(root->right, arr);
    }
};
