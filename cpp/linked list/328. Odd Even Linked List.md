class Solution {
public:
    ListNode* oddEvenList(ListNode* head) {
        if (!head || !head->next) return head;
        ListNode *pre = head;
        ListNode *cur = head->next;
        while (cur) {
            if (cur->next) {
                ListNode *temp = cur->next;
                cur->next = temp->next;
                temp->next = pre->next;
                pre->next = temp;
                cur = cur->next;
                pre = pre->next;
            } else {
                cur = cur->next;
            }
        }
        return head;
    }
};
