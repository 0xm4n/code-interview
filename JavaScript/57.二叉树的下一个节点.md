function TreeLinkNode(x){
    this.val = x;
    this.left = null;
    this.right = null;
    this.next = null;
}

function GetNext(pNode)
{
    if(pNode == null) return null;
    if(pNode.right != null) {
        pNode = pNode.right;
        while(pNode.left !== null){
            pNode = pNode.left;
        }
        return pNode;
    }

    while(pNode.next !== null){
        if(pNode == pNode.next.left){
            return pNode.next;
        }
        pNode = pNode.next;
    }
    return null;
}
