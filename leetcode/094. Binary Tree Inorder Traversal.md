Solution 1 recursion
```cpp


class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> ans;
        inorder(root, ans);
        return ans;
        
    }
private:
    void inorder(TreeNode* root, vector<int>& ans) {
        if (!root) return;
        if (root->left) inorder(root->left, ans);
        ans.push_back(root->val);
        if (root->right) inorder(root->right, ans);
    }
};
```

Solution 2 Iterative

```cpp
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode *> s;
        TreeNode *p = root;
        while (p || !s.empty()) {
            while (p) {
                s.push(p);
                p = p->left;
            }
            p = s.top(); s.pop();
            res.push_back(p->val);
            p = p->right;
        }
        return res;
    }
};
```
