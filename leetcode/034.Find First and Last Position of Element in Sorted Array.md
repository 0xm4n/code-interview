class Solution {
    public int[] searchRange(int[] nums, int target) {
        int[] res = new int[2];
        res[0] = findFirst(nums, target);
        res[1] = findLast(nums, target);
        return res;  
    }
    
    public int findFirst(int[] nums, int target) {
        int first = -1;
        int lo = 0, hi = nums.length;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (nums[mid] < target) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
            if (nums[mid] == target) {
                first = mid;
            }
        }
        return first;
    }
    
    public int findLast(int[] nums, int target) {
        int last = -1;
        int lo = 0, hi = nums.length;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (nums[mid] <= target) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
            if (nums[mid] == target) {
                last = mid;
            }
        }
        return last;
    }
}
